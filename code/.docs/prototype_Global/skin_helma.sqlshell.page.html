129:     function extraEditor(propname, localCols, targetCols) {
130:         var type = opener.document.getElementsByName("maptype_" + propname)[0];
131:         for (var i in mappingOptions) {
132:             var fieldname = propname + '_' + i;
133:             if (!type || parseInt(type.value) < mappingOptions[i]) continue;
134:             var field = opener.document.getElementById(fieldname);
135:             if (field) {
136:                 document.write("<tr><td>" + i + "</td><td>");
137:                 if (i in {foreign: 1, accessname: 1, group:1} && targetCols) {
138:                     writeSelect(fieldname, field.value, targetCols);
139:                 } else if (i == "local" && localCols) {
140:                     writeSelect(fieldname, field.value, localCols);
141:                 } else {
142:                     document.write("<input name=\'" + fieldname + "\' value=\'" + field.value + "\'>");
143:                 }
144:                 document.write("</td></tr>");
145:             } else document.write(fieldname + "<br>");
146:         }

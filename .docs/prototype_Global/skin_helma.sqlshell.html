255:     function extraEditor(propname, localCols, targetCols) {
256:         var type = opener.document.getElementsByName("maptype_" + propname)[0];
257:         for (var i in mappingOptions) {
258:             var fieldname = propname + '_' + i;
259:             if (!type || parseInt(type.value) < mappingOptions[i]) continue;
260:             var field = opener.document.getElementById(fieldname);
261:             if (field) {
262:                 document.write("<tr><td>" + i + "</td><td>");
263:                 if (i in {foreign: 1, accessname: 1, group:1} && targetCols) {
264:                     writeSelect(fieldname, field.value, targetCols);
265:                 } else if (i == "local" && localCols) {
266:                     writeSelect(fieldname, field.value, localCols);
267:                 } else {
268:                     var value = field.value ? field.value.replace('"', "&quot;", "g") : "";
269:                     document.write("<input name=\"" + fieldname + "\" value=\"" + value + "\">");
270:                 }
271:                 document.write("</td></tr>");
272:             } else document.write(fieldname + "<br>");
273:         }
274:     }

276:     function writeSelect(name, value, options) {
277:         document.write("<select name=\'" + name + "\'>");
278:         document.write("<option value=''>");
279:         for (var i in options) {
280:             document.write("<option value=\'" + options[i] + "\'");
281:             document.write(value == options[i] ? " selected=\'selected=\'>" : ">");
282:             document.write(options[i]);
283:             document.write("</option>");
284:         }
285:         document.write("</select>");
286:     }
